[77. 组合](https://leetcode.cn/problems/combinations/)

````
var combine = function (n, k) {
  const res = []

  const path = []

  function combineFunc(n, k, startIndex) {
    if (path.length === k) {
      res.push([...path])
    }
    // 优化：
    // 已经选择的个数：path.length    
    // 还需要的元素个数：k - path.length
    // 剩余元素要大于还需要的个数  n - i >= k - path.length 
    // 包括起始位置，左闭右开 ，所以要 + 1
    // i <= n - k + path.length + 1
    for (let i = startIndex; i <= n; i++) {
      path.push(i)
      combineFunc(n, k, i + 1)
      path.pop()
    }
  }

  combineFunc(n, k, 1)

  console.log(res, '-- res')
  return res
};
````
